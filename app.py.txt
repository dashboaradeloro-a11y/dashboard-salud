import streamlit as st
import pandas as pd
import plotly.express as px

# Configuración de página
st.set_page_config(page_title="Dashboard Abastecimiento", layout="wide")

# --- CONEXIÓN A GOOGLE SHEETS ---
# REEMPLAZA esta URL con la de tu propio Google Sheet
SHEET_URL = "https://docs.google.com/spreadsheets/d/1Tt5BPmaOIPCwg8IAiJ1_RCc11D9ruZwvpiLSHWvAspU/edit?usp=sharing"

@st.cache_data(ttl=600) # Se actualiza cada 10 minutos
def load_data(url):
    # Transformar la URL para descargar como CSV
    csv_url = url.replace('/edit#gid=', '/export?format=csv&gid=')
    return pd.read_csv(csv_url)

try:
    df = load_data(SHEET_URL)
    
    # Limpieza básica de nombres de columnas (quitar espacios o saltos de línea)
    df.columns = [c.strip().replace('\n', ' ') for c in df.columns]

    # --- INTERFAZ ---
    st.title("📊 Control de Abastecimiento en Tiempo Real")

    # Filtros
    col1, col2 = st.columns(2)
    with col1:
        oficina = st.selectbox("OFICINA TECNICA", options=sorted(df['OFICINA TECNICA'].unique()))
    with col2:
        unidad = st.selectbox("UNIDAD OPERATIVA", options=sorted(df[df['OFICINA TECNICA'] == oficina]['UNIDAD OPERATIVA'].unique()))

    # Filtrar datos
    mask = (df['OFICINA TECNICA'] == oficina) & (df['UNIDAD OPERATIVA'] == unidad)
    df_filtered = df[mask]

    # Convertir columna de porcentaje a número por si acaso viene como texto "60,00%"
    if df_filtered['PORCENTAJE DE ABASTECIMIENTO'].dtype == object:
        df_filtered['PORCENTAJE DE ABASTECIMIENTO'] = df_filtered['PORCENTAJE DE ABASTECIMIENTO'].str.replace('%', '').str.replace(',', '.').astype(float)

    # --- MÉTRICA ---
    avg_total = df_filtered['PORCENTAJE DE ABASTECIMIENTO'].mean()
    st.markdown(f"<h3 style='text-align: center;'>PORCENTAJE DE ABASTECIMIENTO TOTAL</h3>", unsafe_allow_html=True)
    st.markdown(f"<h1 style='text-align: center; color: #6f2da8;'>{avg_total:.2f} %</h1>", unsafe_allow_html=True)

    # --- GRÁFICO ---
    # Ordenar las categorías para que coincidan con tu imagen
    orden_categorias = ['IMAGEN', 'LABORATORIO', 'USO GENERAL', 'MEDICAMENTOS', 'ODONTOLOGICOS']
    
    fig = px.bar(
        df_filtered, 
        x='CATEGORIA', 
        y='PORCENTAJE DE ABASTECIMIENTO',
        text='PORCENTAJE DE ABASTECIMIENTO',
        color_discrete_sequence=['#6f2da8'],
        category_orders={"CATEGORIA": orden_categorias}
    )

    fig.update_traces(texttemplate='%{text:.2f}%', textposition='outside')
    fig.update_layout(
        yaxis_range=[0, 110],
        xaxis_title="",
        yaxis_title="%",
        plot_bgcolor='rgba(0,0,0,0)',
    )

    st.plotly_chart(fig, use_container_width=True)

except Exception as e:
    st.error("Error al conectar con Google Sheets. Verifica que el archivo sea público.")
    st.write(e)
